⏺ -- step execution (오늘자)
  UPDATE BATCH_STEP_EXECUTION SET STATUS = 'FAILED', EXIT_CODE = 'FAILED'
  WHERE JOB_EXECUTION_ID IN (
    SELECT je.JOB_EXECUTION_ID FROM BATCH_JOB_EXECUTION je
    JOIN BATCH_JOB_INSTANCE ji ON je.JOB_INSTANCE_ID = ji.JOB_INSTANCE_ID
    WHERE ji.JOB_NAME = ''
      AND je.STATUS = 'ABANDONED'
      AND je.CREATE_TIME >= CURRENT_DATE
  );

  -- job execution (오늘자)
  UPDATE BATCH_JOB_EXECUTION SET STATUS = 'FAILED', EXIT_CODE = 'FAILED'
  WHERE STATUS = 'ABANDONED'
    AND CREATE_TIME >= CURRENT_DATE
    AND JOB_INSTANCE_ID IN (
      SELECT JOB_INSTANCE_ID FROM BATCH_JOB_INSTANCE
      WHERE JOB_NAME = ''
    );


  SELECT je.JOB_EXECUTION_ID, je.STATUS, je.CREATE_TIME
  FROM BATCH_JOB_EXECUTION je
  JOIN BATCH_JOB_INSTANCE ji ON je.JOB_INSTANCE_ID = ji.JOB_INSTANCE_ID
  WHERE ji.JOB_NAME = ''
    AND je.STATUS = 'ABANDONED'
    AND je.CREATE_TIME >= CURRENT_DATE;
